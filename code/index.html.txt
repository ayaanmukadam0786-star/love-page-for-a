    <script>
        // ---- Elements ----
        const yesBtn = document.getElementById("yesBtn");
        const noBtn = document.getElementById("noBtn");
        const message = document.getElementById("message");
        const mainHeading = document.getElementById("mainHeading");
        const mainGif = document.getElementById("mainGif");
        const photoContainer = document.getElementById("photoContainer");
        const bgMusic = document.getElementById('bgMusic');
        const startOverlay = document.getElementById('startOverlay');
        const startBtn = document.getElementById('startBtn');
        const buttonsContainer = document.getElementById('buttons');

        // ---- State ----
        let noClickCount = 0;
        let isRunningAway = false;

        // ---- Audio Start Logic ----
        startBtn.addEventListener('click', () => {
            startOverlay.style.opacity = '0';
            setTimeout(() => {
                startOverlay.style.display = 'none';
            }, 500);
            
            // Play music
            bgMusic.volume = 0.5; 
            bgMusic.play().catch(error => {
                console.log("Audio play failed:", error);
            });
        });

        // ---- YES Button Logic ----
        yesBtn.addEventListener("click", () => {
            // 1. Change Text
            mainHeading.innerText = "Yay! I love you too Anushka! ‚ù§Ô∏è";
            message.innerText = ""; // Clear the "Please say yes" text
            
            // 2. Change Bear GIF to Kissing Bears
            mainGif.src = "https://media1.tenor.com/m/gUiu1zyxfzYAAAAC/bear-kiss-bear-kisses.gif";

            // 3. Show Your Photo
            photoContainer.classList.add("show");

            // 4. Hide Buttons
            buttonsContainer.style.display = "none";

            // 5. Spawn Hearts
            spawnHearts();
            setInterval(spawnHearts, 300);
        });

        // ---- NO Button Logic ----
        function moveNoButton() {
            const maxWidth = window.innerWidth - noBtn.offsetWidth - 20;
            const maxHeight = window.innerHeight - noBtn.offsetHeight - 20;
            const randomX = Math.random() * maxWidth;
            const randomY = Math.random() * maxHeight;

            noBtn.style.position = "fixed";
            noBtn.style.left = randomX + "px";
            noBtn.style.top = randomY + "px";
            
            noBtn.classList.add('spin');
            setTimeout(() => noBtn.classList.remove('spin'), 400);
        }

        noBtn.addEventListener("click", (e) => {
            noClickCount++;
            if (noClickCount === 1) {
                message.innerText = "Will you forgive me? ü•∫";
                mainGif.src = "https://media1.tenor.com/m/K2Q_3q2hk5oAAAAC/bear-cry.gif";
            } else if (noClickCount === 2) {
                message.innerText = "Please na Anushka... üíó";
            } else if (noClickCount === 3) {
                message.innerText = "Please na baccha üò¢";
            } else if (noClickCount === 4) {
                message.innerText = "Okay fine‚Ä¶ but I still love you üò§‚ù§Ô∏è";
            } else if (noClickCount >= 5) {
                message.innerText = "Now you *have* to press YES üò≠üòÇ";
                isRunningAway = true;
                moveNoButton();
            }
        });

        const runAwayHandler = () => {
            if (isRunningAway) {
                moveNoButton();
            }
        };

        noBtn.addEventListener("mouseover", runAwayHandler);
        noBtn.addEventListener("touchstart", (e) => {
            if (isRunningAway) {
                e.preventDefault();
                moveNoButton();
            }
        });

        // ---- Heart Animation ----
        function spawnHearts() {
            for (let i = 0; i < 5; i++) {
                const heart = document.createElement("div");
                heart.classList.add("heart");
                heart.innerHTML = "‚ù§Ô∏è";
                heart.style.left = Math.random() * 100 + "vw";
                heart.style.animationDuration = Math.random() * 2 + 3 + "s";
                document.body.appendChild(heart);
                setTimeout(() => heart.remove(), 5000);
            }
        }
    </script>
